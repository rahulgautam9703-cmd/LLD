public class RoomNotAvailableException extends RuntimeException {
    public RoomNotAvailableException(String roomId) {
        super("Room " + roomId + " is not available for the requested slot");
    }
}

public class UserNotAvailableException extends RuntimeException {
    public UserNotAvailableException(String userName) {
        super(userName + " is not available for the requested slot");
    }
}
